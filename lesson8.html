Lesson 8: Supernatural Identity

Using Boolean Logic and Complex Structs in Mystic Falls

The Concept: Identity Logic

In TVD, a character's "Species" can change. Elena starts as Human, then becomes a Vampire. Klaus is a Hybrid. In C, we use Integers as Booleans (0 or 1) to track these permanent changes.


#include <stdio.h>
#include <string.h>

struct Supernatural {
    char name[30];
    int isVampire;
    int isWerewolf;
    int isWitch;
    int hasDaylightRing;
    int age;
};

void checkSpecies(struct Supernatural s) {
    printf("\n--- Analyzing: %s ---\n", s.name);
    
    if (s.isVampire && s.isWerewolf) {
        printf("Identity: THE HYBRID. High danger level.\n");
    } else if (s.isVampire) {
        printf("Identity: VAMPIRE.\n");
        if (!s.hasDaylightRing) {
            printf("Warning: Stay out of the sun!\n");
        }
    } else if (s.isWitch) {
        printf("Identity: BENNETT WITCH / SPELLCASTER.\n");
    } else {
        printf("Identity: HUMAN (or Doppelgänger).\n");
    }
}

int main() {
    // Initializing the Mystic Falls gang
    struct Supernatural stefan = {"Stefan Salvatore", 1, 0, 0, 1, 171};
    struct Supernatural klaus = {"Niklaus Mikaelson", 1, 1, 0, 1, 1000};
    struct Supernatural elena = {"Elena Gilbert", 0, 0, 0, 0, 18};

    checkSpecies(stefan);
    checkSpecies(klaus);
    checkSpecies(elena);

    // Event: Elena turns!
    printf("\n[PLOT TWIST]: Elena dies with vampire blood in her system...\n");
    elena.isVampire = 1;
    elena.hasDaylightRing = 1;
    
    checkSpecies(elena);

    return 0;
}

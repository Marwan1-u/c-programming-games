Lesson 9: The AAA Game Loop

Merging Physics, Input, and Combat into one Infinite Loop.

⚠️ Warning: This code uses C++ Logic. Ensure your compiler is set to C++ mode! 

#include <iostream>
#include <string>

// 1. DATA STRUCTURE: Managing character states
struct GameCharacter {
    std::string name;
    float health;
    float stamina;
    bool bIsAttacking;
};

// 2. COMBAT ENGINE: The logic that calculates hits
void ProcessCombat(GameCharacter &player, GameCharacter &enemy) {
    if (player.bIsAttacking && player.stamina >= 20.0f) {
        std::cout << "⚔️ Attack Triggered! " << player.name << " swings!" << std::endl;
        
        float damage = 45.0f;
        enemy.health -= damage;
        player.stamina -= 20.0f; // Drain resources

        std::cout << "💥 SUCCESS: " << enemy.name << " Health is now " << enemy.health << std::endl;
        player.bIsAttacking = false; // Reset attack state
    }
}

// 3. THE INFINITE LOOP: The heartbeat of Crimson Desert
int main() {
    GameCharacter hero = {"Macduff", 100.0f, 100.0f, false};
    GameCharacter boss = {"The Serpent", 500.0f, 100.0f, false};

    bool bGameRunning = true;
    int frameCount = 0;

    while (bGameRunning) {
        frameCount++;

        // SIMULATED INPUT: Attack on Frame 5
        if (frameCount == 5) hero.bIsAttacking = true;

        // PHYSICS & COMBAT UPDATE
        ProcessCombat(hero, boss);

        // STAMINA REGEN LOGIC
        if (frameCount % 10 == 0 && hero.stamina < 100) {
            hero.stamina += 10.0f;
            std::cout << "[SYSTEM] Regenerating Stamina..." << std::endl;
        }

        // END CONDITION
        if (boss.health <= 0 || frameCount > 50) {
            std::cout << "--- Simulation Complete ---" << std::endl;
            bGameRunning = false;
        }
    }
    return 0;
}
